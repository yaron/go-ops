package main

import "fmt"

type myConfig struct {
	APIKey string
	Secret string
}

func main() {
	config := myConfig{
		APIKey: "wo9wmfojw4rfwsmef",
		Secret: "so9jfsofjsgsg",
	}
	fmt.Println(config.APIKey)
}

type myError struct { Message string }

func (e myError) Error() string {
	return e.Message
}

func errorFunc() (string, error) {
	return "", myError{Message: "Sorry..."}
}

func main() {
	value, err := errorFunc()
	if err != nil {
		fmt.Println(err)
		return
	}
	fmt.Println(value)
}

type error interface {
    Error() string
}

Building a tool

CLI: https://github.com/urfave/cli
go get github.com/urfave/cli
import (
	"fmt"
	"os"

	"github.com/urfave/cli"
)
func main() {
	app := cli.NewApp()
	err := app.Run(os.Args)
	if err != nil {
		fmt.Println(err)
	}
}
	app.Flags = []cli.Flag{
		cli.StringFlag{
			Name:  "url, u",
			Value: "https://github.com/yaron/go-ops.git",
			Usage: "URL to git clone",
		},
	}
	app.Commands = []cli.Command{
		{
			Name:    "clone",
			Aliases: []string{"c"},
			Usage:   "Clone a git url",
			Action: func(c *cli.Context) error {
				fmt.Printf("URL: %q", c.String("url"))
				return nil
			},
		},
	}
go run cli.go -h
